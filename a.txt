    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        image = findViewById(R.id.imageView);
        image.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Intent intent = new Intent(MediaStore.ACTION_IMAGE_CAPTURE);
                startActivityForResult(intent,113);
            }
        });
    }

    @Override
    protected void onActivityResult(int requestCode, int resultCode, @Nullable Intent data) {
        if(requestCode == 113 && resultCode == RESULT_OK)
        {
            bitmap= (Bitmap) data.getExtras().get("data");

            if(isNetworkAvailable(this)) {
                new UploadFile(image).execute();
            }
            else{
                Toast.makeText(MainActivity.this, "Please check your network", Toast.LENGTH_SHORT).show();
            }
        }
        super.onActivityResult(requestCode, resultCode, data);
    }

    public class UploadFile extends AsyncTask<String, Integer, Boolean> {
        ImageView image;
        ProgressDialog dialog;
        boolean success = false;

        public UploadFile(ImageView image) {
            this.image = image;
        }

        @Override
        protected Boolean doInBackground(String... params) {
            try {
              //  client.setDataTimeout(5000);
                Log.d("vinh", "kkkkkyyy");
                FTPClient client = new FTPClient();
                Log.d("vinh", "rrrrrrr");
                try {
                    client.connect("192.168.1.6", 21);
                }
                catch (Exception e){
                    Log.d("vinh", "fffffff");
                    return false;
                }
                Log.d("vinh", "connect");
                if (client.login("vinh", "vinh@1998")) {

                    client.enterLocalPassiveMode(); // important!
                    client.setFileType(FTP.BINARY_FILE_TYPE);
                    ByteArrayOutputStream out = new ByteArrayOutputStream();
                    bitmap.compress(Bitmap.CompressFormat.PNG,100,out);
                    success = client.storeFile("quatuyetvoi.png", new ByteArrayInputStream(out.toByteArray()));
                    client.logout();
                    client.disconnect();
                    Log.d("vinh", "doInBackground: ");
                }
            } catch (Exception e) {
                Toast.makeText(MainActivity.this, "Unexcepect err", Toast.LENGTH_SHORT).show();
                return false;
            }
            Log.d("vinh", "kkkkkkkkk ");
            return success;
        }

        @Override
        protected void onPreExecute() {
            super.onPreExecute();
            dialog = new ProgressDialog(MainActivity.this);
            dialog.setProgressStyle(ProgressDialog.STYLE_SPINNER);
            dialog.setCancelable(true);
            dialog.setMessage("Uploading...");
            dialog.show();
            Log.d("vinh", "dÄ‘");
        }

        @Override
        protected void onPostExecute(Boolean q) {

            super.onPostExecute(q);
            dialog.dismiss();
            if(q ==true)
            {
                Log.d("vinh", " update success");
                image.setImageBitmap(bitmap);
                Toast.makeText(MainActivity.this, " update success", Toast.LENGTH_SHORT).show();
            }
            else
            {
                Log.d("vinh", " can't update");
                Toast.makeText(MainActivity.this, "can't update", Toast.LENGTH_SHORT).show();
            }
        }
    }
    public static boolean isNetworkAvailable(Context context) {
        return ((ConnectivityManager) context.getSystemService(Context.CONNECTIVITY_SERVICE)).getActiveNetworkInfo() != null;
    }
